#RK4
def rk4(x, y, n, h, f):
    for i in range(n - 1):
        k1 = f(x[i], y[i])
        k2 = f(x[i] + h/2, y[i] + (h * k1)/2)
        k3 = f(x[i] + h/2, y[i] + (h * k2)/2)
        k4 = f(x[i] + h, y[i] + h)

        y[i + 1] = y[i] + h * (k1 + 2*k2 + 2*k3 + k4) / 6

rk4(X, Y_rk4, n, h, f)
