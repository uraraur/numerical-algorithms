f = lambda x: 1/np.cos(x) 
seg = [0, pi/3]
n = 12 #кількість вузлів
X = []
Y = []
c = seg[0] + seg[1] / n
for i in range(n):
    X.append(seg[0] + i * c)
    Y.append(f(X[i]))

print(X)
print(Y)
